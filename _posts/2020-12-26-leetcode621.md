Given a characters array `tasks`, representing the tasks a CPU needs to do, where each letter represents a different task. Tasks could be done in any order. Each task is done in one unit of time. For each unit of time, the CPU could complete either one task or just be idle.

However, there is a non-negative integer `n` that represents the cooldown period between two **same tasks** (the same letter in the array), that is that there must be at least `n` units of time between any two same tasks.

Return *the least number of units of times that the CPU will take to finish all the given tasks*.

```
Input: tasks = ["A","A","A","B","B","B"], n = 2
Output: 8
Explanation: 
A -> B -> idle -> A -> B -> idle -> A -> B
There is at least 2 units of time between any two same tasks.
```



思路：
1. 暴力法 (超时)
```cpp
class Solution {
public:
    int leastInterval(vector<char>& tasks, int n) {
        unordered_map<char, pair<int, int>> records;
        for (char task : tasks) {
            ++records[task].first;
        }
        int ans = 0;
        while (!records.empty()) {
            char task = '#';
            int max_task_count = 0;
            for (auto& record : records) {
                if (record.second.second == 0 && record.second.first > max_task_count) {
                    task = record.first;
                    max_task_count = record.second.first;
                }
                if (record.second.second > 0) {
                    --record.second.second;
                }
            }
            if (task != '#') {
                if (--records[task].first == 0) {
                    records.erase(task);
                }
                else {
                    records[task].second = n;
                }
            }
            ++ans;
        }
        return ans;
    }
};
```

```cpp
class Solution {
public:
    int leastInterval(vector<char>& tasks, int n) {
        int count[26]{};
        int max_freq = 0;
        int max_count = 0;
        for (char task : tasks) {
            ++count[task-'A'];
            if (max_freq == count[task-'A']) {
                ++max_count;
            }
            else if (max_freq < count[task-'A']) {
                max_freq = count[task-'A'];
                max_count = 1;
            }
        }
        int part_count = max_freq - 1;
        int part_len = n - (max_count-1);
        int empty_solts = part_count * part_len;
        int avai_tasks = tasks.size() - max_freq * max_count;
        int idles = max(0, empty_solts - avai_tasks);
        return tasks.size() + idles;
    }
};
```
